ℤ≃ℤᵢ : ℤ ≃ ℤᵢ
ℤ≃ℤᵢ = isoToEquiv (iso to from li ri)
  where
  to : ℤ → ℤᵢ
  to (neg zero)    = pos 0
  to (neg (suc n)) = negsuc n
  to (pos n)       = pos n
  to (0₋≡0₊ _) = pos zero

  from : ℤᵢ → ℤ
  from (pos    n) = pos n
  from (negsuc n) = neg (suc n)

  li : (b : ℤᵢ) → to (from b) ≡ b
  li (pos _)    = refl
  li (negsuc _) = refl

  ri : (a : ℤ) → from (to a) ≡ a
  ri (neg zero)    = sym 0₋≡0₊
  ri (neg (suc _)) = refl
  ri (pos _)       = refl
  ri (0₋≡0₊ i) j = 0₋≡0₊ (i ∨ ~ j)

discreteℤ : Discrete ℤ
discreteℤ = EquivPresDiscrete (invEquiv ℤ≃ℤᵢ) discreteℤᵢ
