private variable
  ℓ : Level
  A B C : Type ℓ

bindₘ=nothing : ∀ {ℓ ℓ′} {A : Type ℓ} {B : Type ℓ′}
                  {f : A → Maybe B} {m : Maybe A}
              → bindₘ m f ＝ nothing
              → (m ＝ nothing) ⊎ (Σ[ x ꞉ A ] (m ＝ just x) × (f x ＝ nothing))
bindₘ=nothing {m = just x}  e = inr (x , refl , e)
bindₘ=nothing {m = nothing} e = inl refl
