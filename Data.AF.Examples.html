T2-inv ((z‚ÇÅ , z‚ÇÇ) , (exz , inl x<z) , (w‚ÇÅ , w‚ÇÇ) , (ezw , inl z<w) , lift ewy) =

T2-inv ((z‚ÇÅ , z‚ÇÇ) , (exz , inl x<z) , (w‚ÇÅ , w‚ÇÇ) , (ezw , inr z<w) , lift ewy) =
  inl       ( ‚â§-<-trans (=‚Üí‚â§ exz) (<-‚â§-trans z<w (=‚Üí‚â§ (ap fst ewy)))
            , <-‚â§-trans x<z (<‚Üí‚â§ z<w ‚àô =‚Üí‚â§ (ap fst ewy)))
T2-inv ((z‚ÇÅ , z‚ÇÇ) , (exz , inr x<z) , (w‚ÇÅ , w‚ÇÇ) , (ezw , inl z<w) , lift ewy) =
  inr $ inl ( <-‚â§-trans x<z (=‚Üí‚â§ (ezw ‚àô ap snd ewy))
            , ‚â§-<-trans (=‚Üí‚â§ exz) (<-‚â§-trans z<w (=‚Üí‚â§ (ap snd ewy))))
T2-inv ((z‚ÇÅ , z‚ÇÇ) , (exz , inr x<z) , (w‚ÇÅ , w‚ÇÇ) , (ezw , inr z<w) , lift ewy) =
  inr $ inr ( ‚â§-<-trans (=‚Üí‚â§ exz) (<-‚â§-trans z<w (=‚Üí‚â§ (ap fst ewy)))
            , <-‚â§-trans x<z (=‚Üí‚â§ (ezw ‚àô ap snd ewy)))
