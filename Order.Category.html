{-# OPTIONS --safe #-}
module Order.Category where

open import Cat.Prelude

open import Order.Base

poset→precategory : ∀ {o ℓ} → Poset o ℓ → Precategory o ℓ
poset→precategory P = cat where
  open Poset P
  open Precategory
  cat : Precategory _ _
  cat .Ob  = P .Ob
  cat .Hom = _≤_
  cat .id  = ≤-refl
  cat ._∘_ = flip ≤-trans
  cat .id-r _      = prop!
  cat .id-l _      = prop!
  cat .assoc _ _ _ = prop!
  cat .Hom-set _ _ = hlevel!
