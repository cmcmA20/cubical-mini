instance
  maybe-is-fin-set : {ℓ : Level} {A : Type ℓ} → ⦃ is-fin-set A ⦄ → is-fin-set (Maybe A)
  maybe-is-fin-set {A} = fin do
    aeq ← enumeration it
    ueq ← enumeration it
    pure $ maybe-as-sum ∙ₑ ⊎-ap ueq aeq ∙ₑ fin-coproduct
