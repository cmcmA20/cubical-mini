-- Quantifiers

record Π-notation {ℓᵃ ℓᵇ ℓ}
  (A : 𝒰 ℓᵃ) ⦃ _ : Underlying A ⦄ (B : 𝒰 ℓᵇ) (R : 𝒰 ℓ) : 𝒰ω where
  field Π : (X : A) (F : ⌞ X ⌟ → B) → R

infixr 6 Π-syntax
Π-syntax
  : {A : Type ℓ} ⦃ u : Underlying A ⦄
    {B : Type ℓ′} {R : Type ℓ″}
    ⦃ p : Π-notation A B R ⦄
    (X : A) (F : ⌞ X ⌟ → B)
  → R
Π-syntax ⦃ p ⦄ = p .Π-notation.Π
syntax Π-syntax X (λ x → F) = Π[ x ꞉ X ] F

-- Automation

instance
  ×-Variadic
    : {A : Type ℓ} {B : Type ℓ′} {R : Type ℓ″}
      {X : Type ℓ‴}
    → ⦃ im : ×-notation A B R ⦄
    → ×-notation (X → A) (X → B) (X → R)
  ×-Variadic ._×_ f g x = f x × g x

  ⊕-Variadic
    : {A : Type ℓ} {B : Type ℓ′} {R : Type ℓ″}
      {X : Type ℓ‴}
    → ⦃ im : ⊕-notation A B R ⦄
    → ⊕-notation (X → A) (X → B) (X → R)
  ⊕-Variadic ._⊕_ f g x = f x ⊕ g x

  ⊎-Variadic
    : {A : Type ℓ} {B : Type ℓ′} {R : Type ℓ″}
      {X : Type ℓ‴}
    → ⦃ im : ⊎-notation A B R ⦄
    → ⊎-notation (X → A) (X → B) (X → R)
  ⊎-Variadic ._⊎_ f g x = f x ⊎ g x

  ⊎₁-Variadic
    : {A : Type ℓ} {B : Type ℓ′} {R : Type ℓ″}
      {X : Type ℓ‴}
    → ⦃ im : ⊎₁-notation A B R ⦄
    → ⊎₁-notation (X → A) (X → B) (X → R)
  ⊎₁-Variadic ._⊎₁_ f g x = f x ⊎₁ g x

  ⊻-Variadic
    : {A : Type ℓ} {B : Type ℓ′} {R : Type ℓ″}
      {X : Type ℓ‴}
    → ⦃ im : ⊻-notation A B R ⦄
    → ⊻-notation (X → A) (X → B) (X → R)
  ⊻-Variadic ._⊻_ f g x = f x ⊻ g x

  ⇒-Variadic
    : {A : Type ℓ} {B : Type ℓ′} {R : Type ℓ″}
      {X : Type ℓ‴}
    → ⦃ im : ⇒-notation A B R ⦄
    → ⇒-notation (X → A) (X → B) (X → R)
  ⇒-Variadic ._⇒_ f g x = f x ⇒ g x

  ¬-Variadic
    : {A : Type ℓ} {R : Type ℓ′}
      {X : Type ℓ″}
    → ⦃ im : ¬-notation A R ⦄
    → ¬-notation (X → A) (X → R)
  ¬-Variadic .¬_ f x = ¬ f x
