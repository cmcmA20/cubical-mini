-- Quantifiers

record Π-notation {ℓa ℓb ℓ}
  (A : 𝒰 ℓa) ⦃ _ : Underlying A ⦄ (B : 𝒰 ℓb) (R : 𝒰 ℓ) : 𝒰ω where
  infixr 6 Π
  field Π : (X : A) (F : ⌞ X ⌟ → B) → R
  syntax Π X (λ x → F) = Π[ x ꞉ X ] F
open Π-notation ⦃ ... ⦄ public
{-# DISPLAY Π-notation.Π _ x f = Π x f #-}
