private variable
  ℓ ℓ′ : Level

  B : Type ℓ′

dec-∥-∥₁-equiv : ∥ Dec P ∥₁ ≃ Dec ∥ P ∥₁
dec-∥-∥₁-equiv = prop-extₑ!
  (∥-∥₁.rec! $ Dec.map pure ∥-∥₁.rec!)
  (Dec.rec (yes <$>_) $ pure ∘ no ∘ _∘ pure)

dec-≃ : A ≃ B → Dec A ≃ Dec B
dec-≃ eqv = dec-as-sum ∙ₑ ⊎-ap (¬-≃ to from) eqv ∙ₑ (dec-as-sum ₑ⁻¹)
  where open Equiv eqv
