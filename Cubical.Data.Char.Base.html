  )

open import Agda.Builtin.String using (primShowChar) public

infix 4 _≈_ _≉_
_≈_ : Rel Char Char _
c₁ ≈ c₂ = toℕ c₁ ≡ toℕ c₂

_≉_ : Rel Char Char _
c₁ ≉ c₂ = ¬ (c₁ ≈ c₂)
