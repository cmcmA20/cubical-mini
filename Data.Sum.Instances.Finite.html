instance
  ⊎-is-fin-set
    : ∀ {ℓ ℓ′} {A : Type ℓ} {B : Type ℓ′}
    → ⦃ is-fin-set A ⦄ → ⦃ is-fin-set B ⦄ → is-fin-set (A ⊎ B)
  ⊎-is-fin-set = fin do
    aeq ← enumeration it
    beq ← enumeration it
    pure $ ⊎-ap aeq beq ∙ₑ fin-coproduct
