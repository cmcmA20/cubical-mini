private variable
  n : HLevel
  ℓ : Level
  A : Type ℓ

instance
  private
    _ : H-Level (suc n) ∥ A ∥₁
    _ = hlevel-prop-instance squash₁

  Bind-∥-∥₁ : Bind (eff ∥_∥₁)
  Bind-∥-∥₁ ._>>=_ = flip rec!

  Lawful-Bind-∥-∥₁ : Lawful-Bind (eff ∥_∥₁)
  Lawful-Bind-∥-∥₁ .Lawful-Bind.>>=-id-l = refl
  Lawful-Bind-∥-∥₁ .Lawful-Bind.>>=-id-r = prop!
  Lawful-Bind-∥-∥₁ .Lawful-Bind.>>=-assoc = prop!
  Lawful-Bind-∥-∥₁ .<*>->>= = prop!
