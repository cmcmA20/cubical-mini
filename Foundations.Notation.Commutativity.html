module _ {ℓᵃ} (A : 𝒰 ℓᵃ) where
  Commutativity : (t : A → A → A) (x y : A) → 𝒰 ℓᵃ
  Commutativity t x y = t y x ＝ t x y

  record Comm ⦃ t : Has-binary-op A ⦄ : 𝒰 ℓᵃ where
    no-eta-equality
    field <>-comm : ∀ x y → Commutativity (t ._<>_) x y
