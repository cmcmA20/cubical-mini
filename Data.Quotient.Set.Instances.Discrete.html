private variable
  ℓ ℓᵃ : Level
  A : Type ℓᵃ
  R : Rel 2 ℓ A

/₂-is-discrete
  : (Rc : is-congruence R)
  → (∀ x y → Dec ⌞ R x y ⌟)
  → is-discrete (A / ⌞ R ⌟ₙ)
/₂-is-discrete {R} congr d = is-discrete-η $ elim₂-prop! λ x y →
  Dec.map (glue/ _ _) (λ f p → absurd (f ((effective {R = R} congr) .fst p))) (d x y)

  decomp-dis-list = decomp (quote /₂-is-discrete) (`meta ∷ `search-under 2 (quote Dec) ∷ [])
