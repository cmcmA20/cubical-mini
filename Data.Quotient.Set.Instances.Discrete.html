private variable
  ℓ ℓᵃ : Level
  A : Type ℓᵃ
  R : Corr 2 ℓ A

/₂-is-discrete
  : (R-c : is-congruence R)
  → (∀ x y → Dec (R x y))
  → is-discrete (A / R)
/₂-is-discrete R-c d = is-discrete-η $ elim₂-prop! λ x y →
  Dec.map (glue/ _ _) (λ f p → absurd (f $ (effective R-c ₑ⁻¹) .fst p)) $ d x y
