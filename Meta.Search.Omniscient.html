omni-prop-helper : ∀{ℓa ℓb} {A : Type ℓa} {B : A → Type ℓb}
                 → (ΣAB-prop : is-prop Σ[ B ])
                 → Decidable B
                 → Omniscient₁ A
                 → Dec Σ[ B ]
omni-prop-helper ΣAB-prop d omn₁ =
  ∥-∥₁.proj (dec-is-of-hlevel 1 ΣAB-prop) $ ∥-∥₁∘dec≃dec∘∥-∥₁ ⁻¹ $ ∃-decision d omn₁
