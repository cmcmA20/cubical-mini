private variable

ℙ : Type ℓ → Type (ℓsuc ℓ)
ℙ X = X → Prop _

private variable A B : ℙ X

subst-∈ : (A : ℙ X) → x ＝ y → x ∈ A → y ∈ A
subst-∈ A = subst (_∈ A)

⊆-refl : (A : ℙ X) → A ⊆ A
⊆-refl _ = id

@0 ℙ-ext : A ⊆ B → B ⊆ A → A ＝ B
ℙ-ext A⊆B B⊆A = fun-ext λ _ → n-ua (prop-extₑ! A⊆B B⊆A)

single : {@(tactic hlevel-tactic-worker) X-set : is-set X} → X → ℙ X
single {X-set} x t = el (x ＝ t) (path-is-of-hlevel′ 1 X-set x t)

infixr 22 _∩_
_∩_ : ℙ X → ℙ X → ℙ X
(A ∩ B) x = el! ((x ∈ A) × (x ∈ B))

infixr 21 _∪_
_∪_ : ℙ X → ℙ X → ℙ X
(A ∪ B) x = el! ((x ∈ A) ⊎₁ (x ∈ B))

⟙ : ℙ X
⟙ _ = el! (Lift _ ⊤)

⟘ : ℙ X
⟘ _ = el! (Lift _ ⊥)

⟘⊆ : ⟘ ⊆ A
⟘⊆ ()

@0 ⊆⟘→⟘ : A ⊆ ⟘ → A ＝ ⟘
⊆⟘→⟘ {A} p = ℙ-ext p (⟘⊆ {A = A})
