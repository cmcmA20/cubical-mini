           ‚Üí f Ôºù g
  Mor-path p q r i = p i , q i , r i

    {-# OVERLAPPING Refl-Hom #-}

    Trans-Hom ._‚àô_ f g = g ‚àò f

    ‚âÖ-Cat-Ob : ‚âÖ-notation Ob Ob (ùí∞ h)
